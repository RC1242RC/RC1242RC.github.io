    implicit none
    double precision, dimension(3), intent(inout) :: i
    integer, intent(in) :: j

        double precision, dimension(3), intent(in) :: y
      end function func
    end interface afunc
    write(*,*) " This is fptrRoutine called with arguments i,j=",i,j,"and a function pointer, func."
